CREATE TABLE CLI(NumCli NUMBER GENERATED BY DEFAULT AS IDENTITY,
  2  NomCli VARCHAR(50) NOT NULL,
  3  Pays VARCHAR(50) NOT NULL, 
  4  Tel NUMBER NOT NULL,
  5  Ville VARCHAR(50)NOT NULL,
  6  Dept VARCHAR(60) NOT NULL,
  7  Nat VARCHAR(50) NOT NULL,
  8  PRIMARY KEY(NumCli));



CREATE TABLE COM(NumCom NUMBER GENERATED BY DEFAULT AS IDENTITY,
NumCli NUMBER,
FraisPort NUMBER NOT NULL,
DateCom DATE NOT NULL,
Payement NUMBER NOT NULL,
PRIMARY KEY(NumCom),
CONSTRAINT fk_CLI_COM
    FOREIGN KEY ( NumCli)
    REFERENCES CLI (NumCli));



CREATE TABLE FOU(NumFou NUMBER GENERATED BY DEFAULT AS IDENTITY,
NomFou VARCHAR(60) NOT NULL,
Pays VARCHAR(50) NOT NULL,
Tel NUMBER NOT NULL,
PRIMARY KEY(NumFou));





CREATE TABLE PRO(NumPro NUMBER GENERATED BY DEFAULT AS IDENTITY, 
NumFou NUMBER,
NomPro  VARCHAR(70),
TypePro VARCHAR(50), 
PrixUnit NUMBER,
PRIMARY KEY(NumPro),
CONSTRAINT fk_PRO_FOU
	 FOREIGN KEY (NumFou)
	 REFERENCES FOU (NumFou));




CREATE TABLE DET(
NumCom NUMBER,
NumPro NUMBER, 
Qte NUMBER NOT NULL, 
Remise NUMBER NOT NULL,
FOREIGN KEY (NumCom)
REFERENCES COM(NumCom),


FOREIGN KEY (NumPro)
REFERENCES PRO(NumPro),

PRIMARY KEY(NumCom,NumPro));
